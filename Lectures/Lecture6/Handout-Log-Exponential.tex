\newcommand{\mytitle}{Exponential and Log Functions}
\myheader{\mytitle}

\begin{itemize}
  \item Derivative of the exponential function:
  $$ \frac{dy}{d}e^t = e^t$$
  \item Derivative of the log function:
 $$\frac{d}{d t} \ln t =\frac{1}{t}$$
\end{itemize}

Find the derivatives for the following functions:\\
\begin{enumerate}
\item $y=e^t$ \\~\\
\item $y=\ln t$ \\~\\
\item $y=ae^{rt}$ \\~\\
\item $y=e^{-t}$ \\~\\
\item $y=\ln a t$ \\~\\
\item $y=\ln t^{c}$ \\~\\
\end{enumerate}
